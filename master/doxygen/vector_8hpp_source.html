642 template <typename T>
643 void
644 vector<T>::_alloc(size_type capacity_new)
645 {
646  assert(pmemobj_tx_stage() == TX_STAGE_WORK);
647  assert(_data == nullptr);
648  assert(_size == 0);
649 
650  if (capacity_new > max_size())
651  throw std::length_error("New capacity exceeds max size.");
652 
653  _capacity = capacity_new;
654 
655  if (capacity_new == 0)
656  return;
657 
658  /*
659  * We need to cache pmemobj_tx_alloc return value and only after that
660  * assign it to _data, because when pmemobj_tx_alloc fails, it aborts
661  * transaction.
662  */
663  persistent_ptr<T[]> res =
664  pmemobj_tx_alloc(sizeof(value_type) * capacity_new,
665  detail::type_num<value_type>());
666 
667  if (res == nullptr)
668  throw transaction_alloc_error(
669  "Failed to allocate persistent memory object");
670 
671  _data = res;
672 }

721 template <typename T>
722 void
723 vector<T>::_grow(size_type count, const_reference value)
724 {
725  assert(pmemobj_tx_stage() == TX_STAGE_WORK);
726  assert(_capacity >= count + _size);
727 
728  pointer dest = _data.get() + static_cast<size_type>(_size);
729  const_pointer end = dest + count;
730  for (; dest != end; ++dest)
731  detail::create<value_type, const_reference>(dest, value);
732  _size += count;
733 }

757 template <typename T>
758 template <typename InputIt,
759  typename std::enable_if<
760  detail::is_input_iterator<InputIt>::value &&
761  std::is_constructible<
762  T,
763  typename std::iterator_traits<
764  InputIt>::reference>::value,
765  InputIt>::type *>
766 void
767 vector<T>::_grow(InputIt first, InputIt last)
768 {
769  assert(pmemobj_tx_stage() == TX_STAGE_WORK);
770  difference_type diff = std::distance(first, last);
771  assert(diff >= 0);
772  assert(_capacity >= static_cast<size_type>(diff) + _size);
773 
774  pointer dest = _data.get() + static_cast<size_type>(_size);
775  _size += static_cast<size_type>(diff);
776  while (first != last)
777  detail::create<value_type>(dest++, *first++);
778 }

798  assert(size_new <= _size);
